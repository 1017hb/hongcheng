src目录下创建 router文件夹  创建文件index.js 
import Vue from 'vue'
import VueRouter from 'vue-router'
//懒加载
const Home = () => from import('../views/home/Home')
Vue.use(VueRouter)

const routes = [
	//默认路径
	{
		path:'',
		redirect:'/home'
	},
	{
		path:'/home'，
		component:Home
	}
	]

const router = new VueRouter({
	routes
})

export defalut router

在main.js中挂载
import router from './router'
